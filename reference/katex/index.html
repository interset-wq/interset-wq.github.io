    \[ \begin{array}{c|ccc} X \backslash Y & y_1 & y_2 & \cdots & y_j & \cdots \\ \hline x_1 & p_{11} & p_{12} & \cdots & p_{1j} & \cdots \\ x_2 & p_{21} & p_{22} & \cdots & p_{2j} & \cdots \\ \vdots & \vdots & \vdots & \vdots & \vdots & \vdots \\ x_i & p_{31} & p_{32} & \cdots & p_{3j} & \cdots \\ \vdots & \vdots & \vdots & \vdots & \vdots & \vdots \\ \end{array} \]
    $$
\begin{array}{c|ccc}

\end{array}
$$
